\section*{Room Key}
\begin{enumerate}
    \item \textbf{The Threshold} (20$\times$30 ft) --- Entry from Sealed Doors, demonic runes, corrupted trickle
    \item \textbf{The Desecrated Gallery} (50$\times$30 ft) --- Offering hall, 3 Lesser Demons, 2 Corrupted Acolytes, tally marks
    \item \textbf{Adrianna's Sanctum} (25$\times$20 ft) --- Serene Exculpator Adrianna (Cleric 8), ritual chamber, throne
    \item \textbf{The Fountain of Demeter} (40 ft diameter) --- Corrupted sacred spring, Sarathek (Greater Demon, HD 9+9), purification quest
    \item \textbf{The Forgotten Stair} (5$\times$5 ft shaft) --- Hidden escape route, 100 ft climb to surface
\end{enumerate}

\section*{Key NPCs}
\begin{itemize}
    \item \textbf{Serene Exculpator Adrianna} (Cleric 8, LE) --- Room 3. Manipulative fallen priestess. Diplomacy first, violence last.
    \item \textbf{Sarathek, Greater Demon} (HD 9+9, CE) --- Room 4. Lord of Gluttony. Regenerates in Fountain. Holy weapons deal double damage.
\end{itemize}
